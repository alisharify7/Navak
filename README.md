# how to run

    windows:
      python -m venv venv

    linux or mac:
      python3 -m venv venv

    
    windows:
      ./venv/Script/activate.bat

    linux or mac:
      source ./venv/lib/activate


    windows:
      pip install -r requirements.txt

    linux or mac:
      pip3 install -r requirements.txt


    windows:
      python app.py

    linux or mac:
      python3 app.py
